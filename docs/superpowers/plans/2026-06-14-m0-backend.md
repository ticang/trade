> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development. Checkbox (`- [ ]`) tracking. 实现与审查分离。

**Goal:** 搭建生产后端骨架与横切地基，使后续 M1（回测/因子）可直接装配。覆盖设计 v0.5 §11 M0 全部范围：项目骨架、配置、SQLite+DuckDB+Repository（多账户）、ProviderRegistry、事件总线、日志、交易日历（调休）、基础数据、PIT 字段、`trading_rule` 表、数据质量验证层。**不含 M0.5**（规则表 v1 录入，独立 plan）。

**Tech Stack:** Python 3.11+ · SQLite(WAL) · DuckDB · exchange_calendars · YAML 配置 · pytest（无新重依赖，仅加 `pyyaml`）

**关联：** 设计 v0.5 §3 架构 · §4.1 数据层 · §5 接口契约 · §6 数据模型 · §7.1/7.2 数据流 · §11 M0

**前置：** main 已集成 UI + 全部 M-1a 探测 + 文档。本计划在 `feat/m0-backend` 分支。

```python
from datetime import datetime
from typing import Protocol

class Clock(Protocol):
    def now(self) -> datetime: ...

class LiveClock:
    def now(self) -> datetime:
        return datetime.now()

class BacktestClock:
    def __init__(self, start: datetime):
        self._t = start
    def now(self) -> datetime:
        return self._t
    def advance(self, dt) -> None:   # 接受 timedelta 或 datetime
        from datetime import timedelta
        self._t = dt if isinstance(dt, datetime) else self._t + dt
```

- [ ] **Step 2: 实现** `quant/data/sqlite_store.py`:
```python
import queue, sqlite3, threading
from contextlib import contextmanager

class SqliteStore:
    def __init__(self, path):
        self._path = path
        self._q: queue.Queue = queue.Queue()
        self._thread: threading.Thread | None = None
        self._conn: sqlite3.Connection | None = None
        self._stop = threading.Event()
        self._flushed = threading.Event()

    def start(self):
        self._conn = sqlite3.connect(self._path, check_same_thread=False)
        self._conn.row_factory = sqlite3.Row
        self._conn.execute("PRAGMA journal_mode=WAL")
        from quant.data.schema import create_sqlite
        create_sqlite(self._conn)
        self._thread = threading.Thread(target=self._loop, daemon=True)
        self._thread.start()

    def _loop(self):
        assert self._conn is not None
        while not self._stop.is_set():
            try:
                task = self._q.get(timeout=0.1)
            except queue.Empty:
                continue
            if task[0] == "exec":
                _, sql, params, done = task
                self._conn.execute(sql, params); self._conn.commit(); done.set()
            elif task[0] == "stop":
                return

    def execute(self, sql, params=()):
        done = threading.Event()
        self._q.put_nowait(("exec", sql, params, done))  # asyncio 侧非阻塞
        return done  # 调用方可等可不等

    def query_one(self, sql, params=()):
        # 读在调用线程（WAL 允许并发读）
        cur = self._conn.execute(sql, params)  # type: ignore[union-attr]
        return cur.fetchone()

    def flush(self):
        # 排空队列：投一个 barrier 等其完成
        done = threading.Event()
        self._q.put_nowait(("barrier", None, None, done))
        done.wait(timeout=5)

    def stop(self):
        self._stop.set()
        self._q.put_nowait(("stop", None, None, None))
        if self._thread: self._thread.join(timeout=5)
```
